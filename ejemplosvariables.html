Ejemplo 3: Factorial de un número

// Programa en C++ que calcula el factorial de un número
#include <iostream>

int main() {
    int numero;
    std::cout << "Ingrese un número: ";
    std::cin >> numero;
    int factorial = 1;
    for(int i = 1; i <= numero; ++i) {
        factorial *= i;
    }
    std::cout << "El factorial de " << numero << " es: " << factorial << std::endl;
    return 0;
}

Ejemplo 4: Números primos

// Programa en C++ que verifica si un número es primo
#include <iostream>

bool esPrimo(int numero) {
    if(numero <= 1) {
        return false;
    }
    for(int i = 2; i <= numero / 2; ++i) {
        if(numero % i == 0) {
            return false;
        }
    }
    return true;
}

int main() {
    int numero;
    std::cout << "Ingrese un número: ";
    std::cin >> numero;
    if(esPrimo(numero)) {
        std::cout << numero << " es primo." << std::endl;
    } else {
        std::cout << numero << " no es primo." << std::endl;
    }
    return 0;
}
